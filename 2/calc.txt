package com.example.calc;

import androidx.appcompat.app.AppCompatActivity;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity {
    TextView ob1,ob2,ob3;
    EditText eob1,eob2,eob3;
    Button bob1,bob2,bob3,bob4;
    String s1,s2,s3;




    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        ob1=(TextView) findViewById(R.id.t1);
        ob2=(TextView) findViewById(R.id.t2);
        ob3=(TextView) findViewById(R.id.t3);
        eob1=(EditText) findViewById(R.id.e1);
        eob2=(EditText) findViewById(R.id.e2);
        eob3=(EditText) findViewById(R.id.e3);
        bob1=(Button) findViewById(R.id.b1);
        bob2=(Button) findViewById(R.id.b2);
        bob3=(Button) findViewById(R.id.b3);
        bob4=(Button) findViewById(R.id.b4);
        bob1.setOnClickListener(new View.OnClickListener()

        {
            @Override
            public void onClick(View view) {
                s1=eob1.getText().toString();
                s2=eob2.getText().toString();
                int a,b,c;
                a=Integer.parseInt(s1);
                b=Integer.parseInt(s2);
                c=a+b;
                s3=String.valueOf(c);
                eob3.setText(s3);
            }

        });
        bob2.setOnClickListener(new View.OnClickListener()

        {
            @Override
            public void onClick(View view) {
                s1=eob1.getText().toString();
                s2=eob2.getText().toString();
                int a,b,c;
                a=Integer.parseInt(s1);
                b=Integer.parseInt(s2);
                c=a-b;
                s3=String.valueOf(c);
                eob3.setText(s3);
            }

        });
        bob3.setOnClickListener(new View.OnClickListener()

        {
            @Override
            public void onClick(View view) {
                s1=eob1.getText().toString();
                s2=eob2.getText().toString();
                int a,b,c;
                a=Integer.parseInt(s1);
                b=Integer.parseInt(s2);
                c=a/b;
                s3=String.valueOf(c);
                eob3.setText(s3);
            }

        });
        bob4.setOnClickListener(new View.OnClickListener()

        {
            @Override
            public void onClick(View view) {
                s1=eob1.getText().toString();
                s2=eob2.getText().toString();
                int a,b,c;
                a=Integer.parseInt(s1);
                b=Integer.parseInt(s2);
                c=a*b;
                s3=String.valueOf(c);
                eob3.setText(s3);
            }

        });



        

    }
}
